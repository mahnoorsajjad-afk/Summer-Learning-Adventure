  function goTo(n) {
    if (n < 0 || n >= TOTAL) return;
    currentSlide = n;
    track.style.transform = `translateX(-${n * 100}vw)`;
    // Update dots
    document.querySelectorAll('.dot').forEach((d,i) => d.classList.toggle('active', i===n));
    // Update nav buttons
    for (let i=0;i<TOTAL;i++) {
      const btn = document.getElementById('nav-'+i);
      if (btn) btn.classList.toggle('active', i===n);
    }

    document.getElementById('arrowPrev').classList.toggle('hidden', n===0);
    document.getElementById('arrowNext').classList.toggle('hidden', n===TOTAL-1);

  const outer = document.querySelector('.carousel-track-outer');

    touchStartX = e.touches[0].clientX;
    touchStartY = e.touches[0].clientY;
    isSwiping = false;
  }, {passive:true});
  outer.addEventListener('touchmove', e => {
    const dx = e.touches[0].clientX - touchStartX;
    const dy = e.touches[0].clientY - touchStartY;
    if (!isSwiping && Math.abs(dx) > Math.abs(dy) && Math.abs(dx) > 10) {
      isSwiping = true;
    }
    if (isSwiping) e.preventDefault();
  }, {passive:false});
  outer.addEventListener('touchend', e => {
    if (!isSwiping) return;
    const dx = e.changedTouches[0].clientX - touchStartX;
    if (Math.abs(dx) > 50) goTo(currentSlide + (dx < 0 ? 1 : -1));
  }, {passive:true});

  function roundRect(ctx,x,y,w,h,r){
    ctx.beginPath();ctx.moveTo(x+r,y);ctx.lineTo(x+w-r,y);ctx.quadraticCurveTo(x+w,y,x+w,y+r);
    ctx.lineTo(x+w,y+h-r);ctx.quadraticCurveTo(x+w,y+h,x+w-r,y+h);
    ctx.lineTo(x+r,y+h);ctx.quadraticCurveTo(x,y+h,x,y+h-r);
    ctx.lineTo(x,y+r);ctx.quadraticCurveTo(x,y,x+r,y);ctx.closePath();ctx.fill();
  }
